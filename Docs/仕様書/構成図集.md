## 3. DB構成図（ER図）

```mermaid
erDiagram
    users {
        BIGSERIAL id PK
        VARCHAR_30 name
        VARCHAR_255 email UK
        VARCHAR_255 password_hash "NULL可"
        VARCHAR_20 oauth_provider "NULL可"
        VARCHAR_255 oauth_subject "NULL可"
        VARCHAR_20 role "PARENT/PARTNER/GRANDPARENT/CHILD"
        TIMESTAMP created_at
        TIMESTAMP updated_at
    }

    children {
        BIGSERIAL id PK
        BIGINT parent_user_id FK
        BIGINT child_user_id FK "NULL可"
        VARCHAR_20 name
        VARCHAR_255 avatar
        VARCHAR_7 theme_color
        DATE birth_date "NULL可"
        TIMESTAMP created_at
        TIMESTAMP updated_at
    }

    transactions {
        BIGSERIAL id PK
        BIGINT child_id FK
        VARCHAR_10 type "income/expense"
        INTEGER amount
        VARCHAR_20 category "allowance/earned/new_year/other/NULL"
        VARCHAR_100 memo "NULL可"
        DATE transaction_date
        TIMESTAMP created_at
    }

    goals {
        BIGSERIAL id PK
        BIGINT child_id FK
        VARCHAR_30 name
        INTEGER target_amount
        DATE target_date "NULL可"
        VARCHAR_20 purpose_category "NULL可"
        VARCHAR_200 message "NULL可"
        VARCHAR_10 emoji
        TIMESTAMP created_at
    }

    family_shares {
        BIGSERIAL id PK
        BIGINT owner_user_id FK
        BIGINT shared_user_id FK "NULL可"
        VARCHAR_255 invited_email
        VARCHAR_20 share_role "PARTNER/GRANDPARENT"
        VARCHAR_20 permission "VIEW_ONLY/EDIT"
        VARCHAR_20 status "PENDING/ACCEPTED/REVOKED"
        VARCHAR_255 invite_token "NULL可"
        TIMESTAMP created_at
        TIMESTAMP updated_at
    }

    users ||--o{ children : "親として登録"
    users |o--o| children : "子供アカウント"
    children ||--o{ transactions : "持つ"
    children ||--o{ goals : "持つ"
    users ||--o{ family_shares : "招待する(owner)"
    users |o--o{ family_shares : "招待される(shared)"
```

## 4. パッケージ構成図（クラス図）

```mermaid
classDiagram
    namespace controller {
        class AuthController {
            +showLogin() String
            +showRegister() String
            +register(RegisterForm) String
            +logout() String
        }
        class DashboardController {
            +dashboard(Model) String
        }
        class ChildController {
            +family(Model) String
            +showNew() String
            +create(ChildForm) String
            +showEdit(Long) String
            +update(Long, ChildForm) String
            +delete(Long) String
        }
        class TransactionController {
            +detail(Long, String, Model) String
            +showNew(Long, String) String
            +create(Long, TransactionForm) String
            +delete(Long, Long) String
            +chartData(Long, String) ResponseEntity
        }
        class GoalController {
            +showNew(Long) String
            +create(Long, GoalForm) String
            +showEdit(Long, Long) String
            +update(Long, Long, GoalForm) String
            +delete(Long, Long) String
        }
        class FamilySharingController {
            +showSharing(Model) String
            +invite(FamilyShareInviteForm) String
            +accept(String) String
            +revoke(Long) String
            +resend(Long) String
        }
    }

    namespace service {
        class DashboardService {
            +getDashboardData(Long) DashboardViewModel
            +getFamilyTotalBalance(Long) int
            +getMonthlyGrowthRate(Long) double

        }
        class ChildService {
            +findAllByParent(Long) List~Child~
            +findById(Long) Child

            +findByChildAndMonth(Long, YearMonth) List~Transaction~
```
